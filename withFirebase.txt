import React, { useState } from "react";
import {
  List,
  ListItem,
  IconButton,
  Box,
  ListItemIcon,
  TextareaAutosize,
} from "@mui/material";
import { DragDropContext, Draggable, Droppable } from "react-beautiful-dnd";
import ArrowUpwardIcon from "@mui/icons-material/ArrowUpward";
import ArrowDownwardIcon from "@mui/icons-material/ArrowDownward";
import ArchiveIcon from "@mui/icons-material/Archive";
import DragHandleIcon from "@mui/icons-material/DragHandle";
import { MAIN_COLOUR, MAIN_LINE_HEIGHT, SECONDARY_FONT_SIZE, TASK_ITEM_MIN_HEIGHT } from "../utils/constants";
import Tickbox from "./Tickbox";
import CheckBoxOutlineBlankIcon from '@mui/icons-material/CheckBoxOutlineBlank';
import CheckBoxIcon from '@mui/icons-material/CheckBox';
import EastIcon from '@mui/icons-material/East';

// Firebase imports
import { getDatabase, ref, update } from "firebase/database";

// Main component
const OrderedList = () => {
  const [items, setItems] = useState([
    { id: "1", text: "Item 1", checked: false, order: 0 },
    { id: "2", text: "Item 2", checked: false, order: 1 },
    { id: "3", text: "Item 3", checked: false, order: 2 },
  ]);

  const [archivedItems, setArchivedItems] = useState([]);

  const db = getDatabase();
  const tasksRef = ref(db, 'global_tasks');

  const saveTasksToFirebase = (updatedItems) => {
    update(tasksRef, { items: updatedItems });
  };

  const handleDragEnd = (result) => {
    if (!result.destination) return;

    const newItems = Array.from(items);
    const [movedItem] = newItems.splice(result.source.index, 1);
    newItems.splice(result.destination.index, 0, movedItem);

    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const moveToTop = (index) => {
    const newItems = [...items];
    const [movedItem] = newItems.splice(index, 1);
    newItems.unshift(movedItem);
    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const moveToBottom = (index) => {
    const newItems = [...items];
    const [movedItem] = newItems.splice(index, 1);
    newItems.push(movedItem);
    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const archiveItem = (index) => {
    const newItems = [...items];
    const [archivedItem] = newItems.splice(index, 1);
    setArchivedItems([...archivedItems, archivedItem]);
    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const toggleCheckbox = (index) => {
    const newItems = [...items];
    newItems[index].checked = !newItems[index].checked;
    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const handleTextChange = (index, value) => {
    const newItems = [...items];
    newItems[index].text = value;
    setItems(newItems);
    saveTasksToFirebase(newItems);
  };

  const tickIcons = [
    <CheckBoxOutlineBlankIcon fontSize="inherit" sx={{ color: MAIN_COLOUR }} />,
    <CheckBoxIcon fontSize="inherit" sx={{ color: MAIN_COLOUR }} />,
    <EastIcon fontSize="inherit" sx={{ color: MAIN_COLOUR }} />,
  ];

  return (
    <Box>
      <DragDropContext onDragEnd={handleDragEnd}>
        <Droppable droppableId="droppable-list">
          {(provided) => (
            <List {...provided.droppableProps} ref={provided.innerRef}>
              {items.map((item, index) => (
                <Draggable key={item.id} draggableId={item.id} index={index}>
                  {(provided, snapshot) => (
                    <ListItem
                      ref={provided.innerRef}
                      {...provided.draggableProps}
                      sx={{
                        display: "flex",
                        alignItems: "center",
                        paddingLeft: 0,
                        paddingRight: 0,
                        background: "white",
                      }}
                    >
                      <div
                        style={{
                          display: "flex",
                          alignItems: "center",
                          borderBottom: `${MAIN_LINE_HEIGHT} solid ${MAIN_COLOUR}`,
                          minHeight: TASK_ITEM_MIN_HEIGHT,
                          width: "-webkit-fill-available",
                        }}
                      >
                        <ListItemIcon
                          {...provided.dragHandleProps}
                          sx={{ cursor: "grab", minWidth: "40px" }}
                        >
                          <DragHandleIcon />
                        </ListItemIcon>

                        <Tickbox
                          checkIndex={0}
                          icons={tickIcons}
                          onClick={() => toggleCheckbox(index)}  // Firebase update on checkbox click
                          style={{ padding: 0, fontSize: "30px" }}
                        />

                        <TextareaAutosize
                          value={item.text}
                          onChange={(e) => handleTextChange(index, e.target.value)}  // Firebase update on text change
                          style={{
                            width: "100%",
                            overflow: "hidden",
                            fontSize: SECONDARY_FONT_SIZE
                          }}
                        />

                        <Box
                          sx={{
                            display: "flex",
                            alignItems: "center",
                            justifyContent: "flex-end",
                          }}
                        >
                          <IconButton
                            edge="end"
                            aria-label="move-to-top"
                            onClick={() => moveToTop(index)}
                          >
                            <ArrowUpwardIcon />
                          </IconButton>
                          <IconButton
                            edge="end"
                            aria-label="move-to-bottom"
                            onClick={() => moveToBottom(index)}
                          >
                            <ArrowDownwardIcon />
                          </IconButton>
                          <IconButton
                            edge="end"
                            aria-label="archive"
                            onClick={() => archiveItem(index)}
                          >
                            <ArchiveIcon />
                          </IconButton>
                        </Box>
                      </div>
                    </ListItem>
                  )}
                </Draggable>
              ))}
              {provided.placeholder}
            </List>
          )}
        </Droppable>
      </DragDropContext>
    </Box>
  );
};

export default OrderedList;
